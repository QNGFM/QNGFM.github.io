562. 壁画(前缀和 枚举 思维）

https://www.acwing.com/problem/content/564/

我的思路

将n个墙段除以2，得到要丢掉的墙段数，然后枚举出n-n/2个连续的墙段评分最大和（使用到前缀和）

暴力枚举最大前缀和，因为长度固定了

我的代码

import java.io.*;
import java.util.Scanner;


public class Main {

    public static void main(String[] args) throws Exception {
        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter writer=new BufferedWriter(new OutputStreamWriter(System.out));
        int T=Integer.parseInt(reader.readLine());
        for(int k=1;k<=T;k++){
            int n=Integer.parseInt(reader.readLine());
            String s=reader.readLine();
            int[] a=new int[s.length()+5];
            int[] sum=new int[s.length()+5];
            for(int i=1;i<=s.length();i++) {
                a[i]=s.charAt(i-1)-'0';
                sum[i]=sum[i-1]+a[i];
            }
            int max=n-n/2;//连续墙段数
            long res=0;
            for(int i=1;i+max-1<=n;i++) {
                if(sum[i+max-1]-sum[i-1]>res) {//找最大的得分
                    res=sum[i+max-1]-sum[i-1];
                }
            }	
            writer.write(String.format("Case #%d: %d%n", k,res));
        }
        writer.flush();

    }
}
