试除法求所有约数

模板
vector<int> get_divisors(int x)
{
    vector<int> res;
    for (int i = 1; i <= x / i; i ++ )
        if (x % i == 0)
        {
            res.push_back(i);
            if (i != x / i) res.push_back(x / i);
        }
    sort(res.begin(), res.end());
    return res;
}
例题
869. 试除法求约数
import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        Scanner scanner=new Scanner(System.in);
        int n=scanner.nextInt();
        while(n--!=0) {
            long x=scanner.nextLong();
            getDivisors(x);
        }
        scanner.close();
    }

    public static void getDivisors(long x) {
        List<Long> res=new ArrayList<Long>();
        for(long i=1;i*i<=x;i++) {
            if(x%i==0) {
                res.add(i);
                if(i!=x/i) res.add(x/i);
            }
        }
        Collections.sort(res);  // 对结果列表进行排序
        
        for (Long divisor : res) {  // 迭代并打印排序后的因数列表
            System.out.print(divisor+" ");
        }
        System.out.println();
    }

}

约数个数和约数之和

基本思想

约数个数

第一步：求出质因子及其指数

第二步：造约数，一个质因子的指数选法取决于它的指数+1（加1是指可以取0）

第三步：使用乘法原理，将各质因子指数选法相乘，即可得出约数个数

约数之和

通过一些穷举来进行理解公式

第一步：求出质因子及其指数

第二步：按质因子分类完，通过相加，提取公因式可得 (p1^0 + p1^1 + … + p1^c1) * … * (pk^0 + pk^1 + … + pk^ck)

模板
如果 N = p1^c1 * p2^c2 * ... *pk^ck
约数个数： (c1 + 1) * (c2 + 1) * ... * (ck + 1)
约数之和： (p1^0 + p1^1 + ... + p1^c1) * ... * (pk^0 + pk^1 + ... + pk^ck)

易忘点：最后判断质因子

if(x>1) primes.put(x, primes.getOrDefault(x, 0)+1);

import java.util.HashMap;
import java.util.Map;
import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        Scanner scanner=new Scanner(System.in);
        int MOD=(int) (1e9+7);
        int n=scanner.nextInt();
        Map<Integer, Integer> primes=new HashMap<Integer, Integer>();
        while(n--!=0) {
            int x=scanner.nextInt();
            
            for(int i=2;i<=x/i;i++) {
                while(x%i==0) {
                    x/=i;
                    primes.put(i, primes.getOrDefault(i, 0)+1);
                }
            }
            if(x>1) primes.put(x, primes.getOrDefault(x, 0)+1);
        }
        long res=1;
        for(Map.Entry<Integer, Integer> entry:primes.entrySet()) {
            res = res*(entry.getValue()+1)%MOD;
        }
        System.out.print(res);
        scanner.close();
    }

}

import java.util.HashMap;
import java.util.Map;
import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        Scanner scanner=new Scanner(System.in);
        int MOD=(int) (1e9+7);
        int n=scanner.nextInt();
        Map<Integer, Integer> primes=new HashMap<Integer, Integer>();
        while(n--!=0) {
            int x=scanner.nextInt();
            
            for(int i=2;i<=x/i;i++) {
                while(x%i==0) {
                    x/=i;
                    primes.put(i, primes.getOrDefault(i, 0)+1);
                }
            }
            if(x>1) primes.put(x, primes.getOrDefault(x, 0)+1);
        }
        long res=1;
        for(Map.Entry<Integer, Integer> entry:primes.entrySet()) {
            int a=entry.getValue();
            int p=entry.getKey();
            long t=1;
            while(a--!=0){
                t=(t*p+1)%MOD;
            }
            res = res*t%MOD;
        }
        System.out.print(res);
        scanner.close();
    }

}

欧几里得算法（辗转相除法）

基本思想

模板
int gcd(int a, int b)
{
    return b ? gcd(b, a % b) : a;
}
例题
872. 最大公约数
import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        Scanner scanner=new Scanner(System.in);
        int n=scanner.nextInt();
        while(n--!=0) {
            int a=scanner.nextInt(),b=scanner.nextInt();
            System.out.println(gcb(a,b));
        }
        scanner.close();

    }

    private static int gcb(int a, int b) {
        return b!=0?gcb(b, a%b):a;
    }

}
