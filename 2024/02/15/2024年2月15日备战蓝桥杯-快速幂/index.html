acwing的思路

关于输出

答案是求逆元，a*b^-1

因此，为了保证不出现无解的情况，额外增加限制条件 yi−xi≠998244353（如不增加此条件，则可能出现无解情况，此为比赛原题考虑不周）。如果= 就会出现（c*p ）mod p == 0

关于数学期望

答案有无限多种路径

X(i)表示从高度i开始爬，爬到树顶的事件。求E(x(0))

写出集合X式子

加1表示要Xi要向上走一步，即集合里每个路径加1

根据期望性质，列出期望式子

由于E(x(0))是我们要求的，即上面的f0（因为前面假设了fi=E(xi)），即gn，所以最终输出的答案是gn

关于逆元运算

逆元运算，不是-1次方

费马定理

所以最终代码体现是

acwing的代码

import java.util.Scanner;

public class Main {
    static int P=998244353;
    public static void main(String[] args) {

        Scanner scanner=new Scanner(System.in);
        int n=scanner.nextInt();
        long res=0;//即g0
        while(n--!=0) {
            int x=scanner.nextInt(),y=scanner.nextInt();
            res = (res+1)*y%P*qmi(y-x,P-2)%P;
        }
        System.out.print(res);
        scanner.close();
    }

    public static long qmi(int a, int b) {
        long res=1;
        long t=a;
        while(b>0) {
            if((b&1)==1) res=res*t%P;
            t=t*t%P;
            b>>=1;
        }
        return res;
    }

}

416. 麦森数(高精度,快速幂,数学)

416. 麦森数 - AcWing题库

我的思路

快速幂求解然后转化为字符串统计长度，再输出后50位

需要使用大整数来存储

结果：其他都正确，就是最后一位数不正确，感觉是java与C++的高精度差异出了问题

我的代码

import java.math.BigInteger;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner=new Scanner(System.in);
        int k=scanner.nextInt();
        BigInteger res=qmi(2,k);
        String s=String.valueOf(res);
        int size=s.length();
        System.out.println(size);
        int o=500;
        int c=1;
        if(size<o) {
            for(int i=1;i<=o-size;i++,c++) {
                System.out.print(0);
                if(c==50) {
                    c=0;
                    System.out.println();
                }
            }
            
            for(int i=0;i<size;i++,c++) {
                System.out.print(s.charAt(i));
                if(c==50) {
                    c=0;
                    System.out.println();
                }
            
            }
        }
        else {
            for(int i=size-o;i<size;i++,c++) {
                System.out.print(s.charAt(i));
                if(c==50) {
                    c=0;
                    System.out.println();
                }
            
            }
        }
        
        scanner.close();
    }

    public static BigInteger qmi(int a,int k) {
        BigInteger res=BigInteger.ONE;
        BigInteger t=BigInteger.valueOf(a);
        while(k>0) {
            if((k&1)==1) res=res.multiply(t);
            k>>=1;
            t=t.multiply(t);
        }
        return res;
    }
}

acwing的思路

acwing的C++代码

#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <cmath>

using namespace std;

const int N = 510;

int a[N], c[N], res[N];

void mul(int a[N], int b[N], int res[N])
{
    memset(c, 0, N * 4);
    for (int i = 0; i < 500; i ++ )
        for (int j = 0; j < 500; j ++ )
            if (i + j < 500)
                c[i + j] += a[i] * b[j];
    for (int i = 0, t = 0; i < 500; i ++ )
    {
        t += c[i];
        res[i] = t % 10;
        t /= 10;
    }
}

void qmi(int p)
{
    res[0] = 1;
    a[0] = 2;
    while (p)
    {
        if (p & 1) mul(a, res, res);
        mul(a, a, a);
        p >>= 1;
    }
    res[0] -- ;
}


int main()
{
    int p;
    scanf("%d", &p);
    printf("%d\n", int(log10(2) * p) + 1);

    qmi(p);
    for (int i = 499, j = 0; j < 10; j ++ )
    {
        for (int k = 0; k < 50; k ++, i -- )
            printf("%d", res[i]);
        puts("");
    }
    return 0;
}

acwing的java代码

import java.util.Scanner;

public class Main {
    static int N=510;
    static int[] res=new int[N];
    public static void main(String[] args) {
        Scanner scanner=new Scanner(System.in);
        int p=scanner.nextInt();
        System.out.println((int)(p*Math.log10(2))+1);
        qmi(p);
        for(int i=499,j=0;j<10;j++) {
            for(int k=0;k<50;k++,i--) {
                System.out.print(res[i]);
            }
            System.out.println();
        }
        scanner.close();
    }

    public static void qmi(int p) {
        res[0]=1;
        int[] a=new int[N];
        a[0]=2;
        while(p>0) {
            if((p&1)==1) mul(a,res,res);//res=res*a
            p>>=1;
            mul(a, a, a);//a=a*a
        }
        res[0]--;//因为是计算2^p-1
    }

    public static void mul(int[] a, int[] b, int[] res) {
        int[] c=new int[N*4];
        for(int i=0;i<500;i++) {
            for(int j=0;j<500;j++) {
                c[i+j]+=a[i]*b[j];
            }
        }
        int t=0;
        for(int i=0;i<500;i++) {
            t+=c[i];
            res[i]=t%10;
            t/=10;
        }
    }
}
