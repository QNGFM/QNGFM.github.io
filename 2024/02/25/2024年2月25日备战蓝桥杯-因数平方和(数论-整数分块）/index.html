4662. 因数平方和(数论 整数分块）

https://www.acwing.com/problem/content/4665/

我的思路

除以 1e9+7 的余数，即mod 1e9+7

暴力：遍历求f(n)，累加，过程中进行mod

结果不对，原因不详=>把t也变成long long t=(x/i)%MOD;

我的代码

import java.util.Scanner;

public class Main {
    
    static int MOD=(int)1e9+7;

    public static void main(String[] args) {
        Scanner scanner=new Scanner(System.in);
        int n=scanner.nextInt();
        long sum=0;
        for(int i=1;i<=n;i++) {
            sum=(sum+getFn(i))%MOD;
        }
        System.out.print(sum);
        scanner.close();
    }

    public static long getFn(int x) {
        long temp=0;
        for(long i=1;i<=x/i;i++) {
            if(x%i==0) {
                temp=(temp+i*i)%MOD;
                if(i!=x/i) {
                    long t=(x/i)%MOD;
                    temp=(temp+t*t)%MOD;
                }
            }
        }
        return temp;
    }

}

acwing的代码

import java.util.Scanner;

public class Main {
    
    static int MOD=(int)1e9+7;
    static int INV = 166666668;

    public static void main(String[] args) {
        Scanner scanner=new Scanner(System.in);
        int n=scanner.nextInt();
        long sum=0;
        for(int i=1;i<=n;) {
            int x=n/i,y=n/x;//这个除法已经向下取整
            sum=(sum+x*(calc(y)-calc(i-1)))%MOD;//从i到y的平方和
            i=y+1;//开始下一段
        }
        System.out.print((sum+MOD)%MOD);//因为有减法
        scanner.close();
    }

    public static long calc(int n) {
        return (long)n*(n+1)%MOD*(2*n+1)%MOD*INV%MOD;//除6变成了乘逆元
    }

}
